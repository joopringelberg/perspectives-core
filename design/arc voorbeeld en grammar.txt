GRAMMATICA
ContextDeclaration = ContextType ContextIdentifier "heeft"
ContextType = "Domein" | "Partij" | "Zaak" | "Activiteit" | "Toestand"
ContextDefinition = ContextDeclaration
	Properties
	Rollen
	Acties
Properties = "properties" Property+
Property = ("intern" | "extern") PropertyIdentifier "(" PropertyType "," IsVerplicht, IsFunctioneel ")"
PropertyType = "String" | "Boolean" | "Number" | "Date" | "Memo"
IsVerplicht = "verplicht" | "niet verplicht"
IsFunctioneel = "functioneel" | "niet functioneel"
Views = "views" View+
View = ("intern" | "extern") ViewIdentifier "met properties" PropertyIdentifier+
Rollen = "rollen" Rol+
Rol = RolDeclaration "heeft"
	RolProperties
	Views
RolDeclaration =
	RolType |
	RolType "gevuld door" (RolIdentifier | ContextIdentifier | Query | ContextDeclaration)
	RolIdentifier "gevuld door" (RolIdentifier | ContextIdentifier | Query | ContextDeclaration)
RolProperties = "properties" RolProperty+
RolProperty = PropertyIdentifier "(" PropertyType "," IsVerplicht, IsFunctioneel ")"
RolType = "UserRol" | "DingRol"
Acties = "acties" Actie+
Actie = 
	(Actor | RolIdentifier) (InformatieWerkwoord | WerkwoordIdentifier) |
	(Actor | RolIdentifier) (InformatieWerkwoord | WerkwoordIdentifier) RolIdentifier
	(Actor | RolIdentifier) (InformatieWerkwoord | WerkwoordIdentifier) ViewIdentifier "van" RolIdentifier	
InformatieWerkwoord = "beheert" | "raadpleegt" | "creeÃ«rt" | "verwijdert"

1. Rol en Actie kunnen compacter opgeschreven worden met optionele componenten, maar deze presentatie geeft goed de verschillende stadia weer.
