{- NOTA BENE. Dit is een CRL beschrijving van de interne representatie van de tussenrepresentatie.
Er wordt nooit zo'n bestand, als tekst, ingetikt of gegenereerd!
Ik gebruik deze tekst om een beeld op te bouwen van de interne representatie die 
ontstaat als je een ARC tekst parseert.
M.n. de Context en import statements worden natuurlijk niet gemaakt. Ik heb ze hier nodig
om de eronder staande contexten van namespaces te voorzien.
Evenmin is de GepresenteerdeTekst zelf onderdeel van zo'n tekst.
Alle GepresenteerdeContextTypen zijn in lexicale context gedefinieerd van de GepresenteerdeTekst.
-}

Context usr:"representatief voorbeeld van ARC" als t:

import Context model:Politie$ als pol:

-- Let op: deze context is dus geen onderdeel van zichzelf, zou niet in deze tekst voorkomen!
psp:GepresenteerdeTekst usr:"representatief voorbeeld van ARC"
	extern psp:inlineCommentaar = "Deze regel leidt tot de psp:GepresenteerdeTekst instantie."
	psp:commentaar =>
		psp:volgnummer = 1
		psp:tekst = "Deze tekst heeft alle mogelijke expressies die de ARC syntax toelaat.\n"
		psp:indentering = 0
		psp:aantalTekstRegels = 2
		psp:isMultiLine = true
	psp:omhullendeContext => model:Politie$
		psp:volgnummer = 2
	psp:commentaar =>
		psp:volgnummer = 3
		psp:indentering = 0
	psp:sectie =>
		psp:volgnummer = 4
		psp:sectieNaam = "zaken"
	psp:commentaar =>
		psp:volgnummer = 5
		psp:tekst = "\nDit commentaar staat boven de zaak Aangifte."
		psp:indentering = 0
		psp:aantalTekstregels = 2
	psp:gepresenteerdeContext => t:1
		psp:volgnummer = 6

Sectie psp:gepresenteerdeContext

	extern psp:aantalTekstRegels = 26

				psp:volgnummer = 1

				psp:volgnummer = 3

			psp:rol =>
				psp:GepresenteerdContextType t:6
					psp:context => pol:Aangifte$UserRol$adres
				psp:volgnummer = 4

		psp:contextRolNaam = "psp:rolInContext"
	psp:rol =>

				psp:GepresenteerdContextType t:8
					psp:context => pol:Aangifte$Start$Intake

						psp:sectieNaam = "rollen"

						psp:GepresenteerdContextType t:9
							psp:context => pol:Aangifte$Start$Intake$Uitvoerder

			psp:sectie =>
				psp:volgnummer = 2
